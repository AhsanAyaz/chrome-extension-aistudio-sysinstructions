**Export: `scheduleFlush`**:
```typescript
export function scheduleFlush(): void {
  // Clear resets the 30s window — a flurry of LS_CHANGED calls collapses to one alarm.
  // Hard Rule 3 (PUSH-07): debounce via alarms, not setTimeout (SW-kill unsafe).
  chrome.alarms.clear(FLUSH_ALARM_NAME, () => {
    chrome.alarms.create(FLUSH_ALARM_NAME, { delayInMinutes: 0.5 }); // 30s (Chrome 120+ min)
  });
}
```

**Helper: `buildLastPushedEntry`** — compute hashes for the snapshot:

async function buildLastPushedEntry(
  uuid: string,
  title: string,
  updatedAt: number,
  bodyJson: string,
): Promise<[string, LastPushedEntry]> {  // [uuid, entry]
  const [titleHash, bodyHash] = await Promise.all([
    shortHash(title),
    shortHash(bodyJson),
  ]);
  return [uuid, { titleHash, bodyHash, updatedAt }];
}

**Helper: `writeLastPushed`** — build and persist the snapshot from the flushed batch:
```typescript
async function writeLastPushed(batch: Record<string, unknown>): Promise<void> {
  const registry = batch[REGISTRY_KEY] as SyncRegistry | undefined;
  if (registry === undefined) return;

  const entries = await Promise.all(
    Object.entries(registry)
      .filter(([, rec]) => rec.deletedAt === null)
      .map(([uuid, rec]) => {
        // Reconstruct bodyJson from batch chunk keys for this UUID
        const chunkKeys: string[] = [];
        for (let i = 0; i < rec.chunks; i++) {
          chunkKeys.push(`${BODY_KEY_PREFIX}${uuid}:c${i}`);
        }
        const bodyJson = chunkKeys.map((k) => (batch[k] as string | undefined) ?? '').join('');
        return buildLastPushedEntry(uuid, rec.title, rec.updatedAt, bodyJson);
      }),
  );

  const snapshot: LastPushedSnapshot = {};
  for (const [uuid, entry] of entries) {
    snapshot[uuid] = entry;
  }
  await chrome.storage.local.set({ [LAST_PUSHED_KEY]: snapshot });
}
```

**Helper: `removeStaleBodyKeys`** — clean up old chunk keys when chunk count decreased:
```typescript
async function removeStaleBodyKeys(batch: Record<string, unknown>): Promise<void> {
  const registry = batch[REGISTRY_KEY] as SyncRegistry | undefined;
  if (registry === undefined) return;

  // Read the current (pre-flush) registry from sync to find old chunk counts.
  const r = await chrome.storage.sync.get(REGISTRY_KEY);
  const oldRegistry = (r[REGISTRY_KEY] as SyncRegistry | undefined) ?? {};

  const stale: string[] = [];
  for (const [uuid, newRec] of Object.entries(registry)) {
    const oldRec = oldRegistry[uuid];
    if (oldRec !== undefined && oldRec.chunks > newRec.chunks) {
      for (let i = newRec.chunks; i < oldRec.chunks; i++) {
        stale.push(`${BODY_KEY_PREFIX}${uuid}:c${i}`);
      }
    }
  }
  if (stale.length > 0) {
    await chrome.storage.sync.remove(stale);
  }
}
```

**Export: `flushPendingWrite`** — called by the onAlarm listener in index.ts:
```typescript
export async function flushPendingWrite(): Promise<void> {
  const batch = await drainPendingWrite();
  if (batch === null || Object.keys(batch).length === 0) return; // no-op

  await writeSyncStatus({ state: 'syncing', lastSyncAt: 0 });

  // Remove stale body keys BEFORE writing new registry (avoids ghost chunks).
  await removeStaleBodyKeys(batch);

  try {
    // PUSH-03: single batched set — never a per-item loop.
    await chrome.storage.sync.set(batch);

    const now = Date.now();
    await writeLastPushed(batch);   // D-12: snapshot for next diff cycle (must be AFTER set)
    await clearPendingWrite();       // clear sentinel
    await writeSyncStatus({ state: 'idle', lastSyncAt: now });
    await chrome.action.setBadgeText({ text: '' }); // clear error badge if any
  } catch (err) {
    const msg = String(err);
    if (msg.includes('MAX_WRITE_OPERATIONS_PER_MINUTE') || msg.includes('RATE_LIMIT')) {
      await setErrorState('RATE_LIMITED', msg);
      await chrome.action.setBadgeText({ text: '!' });
      await chrome.action.setBadgeBackgroundColor({ color: '#F59E0B' }); // amber
      // Retry after 60s — pendingWrite stays in local storage.
      chrome.alarms.create(FLUSH_ALARM_NAME, { delayInMinutes: 1 });
    } else if (msg.includes('QUOTA_BYTES')) {
      await setErrorState('QUOTA_EXCEEDED', msg);
      await chrome.action.setBadgeText({ text: '!' });
      await chrome.action.setBadgeBackgroundColor({ color: '#EF4444' }); // red
      // No retry — quota requires user action.
    } else {
      await setErrorState('STRICT_VALIDATION_FAIL', msg);
      await chrome.action.setBadgeText({ text: '!' });
      await chrome.action.setBadgeBackgroundColor({ color: '#EF4444' }); // red
    }
    // pendingWrite is intentionally NOT cleared on failure — preserved for retry/inspection.
  }
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| chrome.storage.local → alarm-flush (pendingWrite read) | Local storage is trusted; no external input at this boundary |
| alarm-flush → chrome.storage.sync (write) | chrome.storage.sync write crosses the Chrome sync boundary |
| chrome.storage.sync error → alarm-flush error handler | Error message string from Chrome API; inspected for known substrings |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-03-a | DoS | Rate limit error path | mitigate | Retry alarm scheduled at delayInMinutes: 1; pendingWrite preserved for retry; badge set to amber so user sees the state |
| T-03-03-b | DoS | Quota exceeded | accept | No retry — quota requires user action (Phase 5 popup will surface delete option). Badge set to red immediately. |
| T-03-03-c | Tampering | Per-item write loop | mitigate | PUSH-03 enforced by design: single `chrome.storage.sync.set(batch)` call; no loop anywhere in flushPendingWrite |
| T-03-03-d | Tampering | writeLastPushed before sync.set | mitigate | Linear success path: `set` → `writeLastPushed` → `clearPendingWrite`. writeLastPushed is never called before the sync.set resolves. |
| T-03-03-e | Integrity | Stale body keys from chunk reduction | mitigate | removeStaleBodyKeys reads old registry and removes extra keys before sync.set; prevents ghost chunks confusing reassembly |
